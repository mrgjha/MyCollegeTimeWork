using System;

 namespace weightrecord
 {
     public class student
     {
         public int weight;
     }



    class studentweight
    {
       public void Run()
       {
           // create an integer
           int x = 50;

           // create a second integer
           int y = 40;

           // display the two integers
           Console.WriteLine("x: {0} y: {1}",
                              x,     y);

           // modify the second integer
           y = 60;

           // display the two integers
           Console.WriteLine("x: {0} y: {1}",
                              x,     y);

           // create a student
           student rana = new student();

           // assign a value to weight
           rana.weight = 45;

           // create a second reference to the student
           student nitu = rana;

           // display their values
           Console.WriteLine("Rana: {0}, Nitu: {1}",
               rana.weight, nitu.weight);

           // assign a new weight to the second reference
           nitu.weight = 70;

           // display the two values
           Console.WriteLine("Milo: {0}, fido: {1}",
               rana.weight, nitu.weight);
       }

       static void Main()
       {
         studentweight s = new studentweight();
          s.Run();
       }
    }
 }
