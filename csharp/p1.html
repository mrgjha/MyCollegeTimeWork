//Java
import java.lang.*;

public class PrintArgs {

public static void main(String[] args) {

for (int i=0;i < args.length;i++)
System.out.println(args[i]);

}
}

//C#

using System;

public class PrintArgs {
public static void Main(string[] args) {
for (int i=0;i < args.Length;i++)
Console.WriteLine(args[i]);

}
}
