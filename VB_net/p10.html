Module Module1

    Sub Main()
        Dim i As Integer
        Dim sum As Integer = 0

        For i = 1 To 100
            sum = sum + i
            Console.Write(i)
            Console.Write("   ")
            Console.WriteLine(sum)
        Next i

        Console.Write("Sum of 1 to 100 is ")
        Console.WriteLine(sum)
        Console.WriteLine("Hit the Return key to exit.")
        Console.ReadLine()
    End Sub

End Module
