Syntax:
Syntax:
For Each element In group
     [statements]
     [Exit For]
     [statements]
 Next element
E.g.
Sub Main()
      Dim intArray(2), intItem As Integer
      intArray(0)=0
      intArray(1)=1
      intArray(2)=2
      For Each intItem In intArray
          System.Console.WriteLine(intArray)
      Next intItem
 End Sub
