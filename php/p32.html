<h1>Writing a file in PHP</h1> 

<?php 
// Open the file for writing 

// Using relative-file name "test.dat" the file will 
// be located in the same directory as this script 

// Using an absolute file name "/test.dat" locats the file in c:\ 
// in Windows XP. 

// This needs to be fixed on Linux since the directory containing 
// the file needs to have read/write privileges. 

$out_file = @fopen("test.dat", "w") or die("Cannot open test.dat");   

fwrite($out_file, "This is line 1" . "\n"); 
fwrite($out_file, "This is line 2" . "\n"); 
fwrite($out_file, sin(1.0) . "\n"); 
fwrite($out_file, sin(2.0) . "\n"); 

// Close the output file 

fclose($out_file); 
?>
