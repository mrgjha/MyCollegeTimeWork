//page1.php

<?php 
   // ini_set("session.cookie_lifetime", "30"); 
   session_start(); 

   // if a background color has not been set display a form 
   // with a text field to enter one, else register the 
   // color as a session variable. 

   if (isset($_REQUEST['bgcolor'])) 
   { 
      $_SESSION['bgcolor'] = $_REQUEST['bgcolor']; 
      display_page_one(); 
   } 
   else 
   { 
      display_form_page(); 
   } 
?> 

<?php 
function display_form_page() 
{ 
   $self = $_SERVER['PHP_SELF']; 
?> 
   <html> 
   <head><title>Session bgcolor page 1</title></head> 
   <body> 
   <h1>Session bgcolor page 1</h1> 
   <form method="post" action="<?php echo $self?>?<?php echo SID?>"> 
   Enter color code as <code>#HHHHHH</code>:  
   <input type="text" name="bgcolor" value=""> 
   <input type="submit" value="Submit Color"> 
   </form> 
   </body> 
   </html> 
<?php 
}    
?> 

<?php 
function display_page_one() 
{ 
?> 
   <html> 
   <head><title>Session bgcolor page 1</title></head> 
   <body bgcolor="<?php echo $_REQUEST['bgcolor']?>"> 
   <h1>Session bgcolor page1</h1> 
   Background color is <?php $_SESSION['bgcolor']?> 
   <p><a href="page2.php?<?php echo SID?>">Go to page 2</a></p> 
   </body> 
   </html> 
<?php 
} 
?> 

//page2.php

<?php 
   // Retrieve the value of the session variable and display a 
   // page showing its value. Then unset the variable so it will 
   // be undefined on the next page. 
    
   // ini_set("session.cookie_lifetime", "30"); 
   session_start(); 
   $bgcolor = isset($_SESSION['bgcolor']) ? $_SESSION['bgcolor'] : "#FFFFFF"; 
   unset($_SESSION['bgcolor']); 
?> 

<html> 
<head><title>Session bgcolor page 2</title></head> 
<body bgcolor="<?php echo $bgcolor ?>"> 
<h1>Session bgcolor page 2</h1> 
<?php echo "Background color is $bgcolor" ?> 
<p><a href = "page3.php?<?php echo SID?>">Go to page 3</a></p> 
</body> 
</html>
