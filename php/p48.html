<?php
$a=true;

if ($a) {
function foo () {
echo "I will not exist if $a evaluates to false!";
}
}
/* need to check $a before calling foo() */
if ($a)
{
  foo();
  }
?>

<?php
function foo() {
function bar() {
echo "I don't exist until foo() is called.\n";
}
}
/* We can't call bar() yet since it doesn't exist. */
foo();
/* Now we can call bar(), foo()'s processesing has
made it accessible. */
bar();
?>
