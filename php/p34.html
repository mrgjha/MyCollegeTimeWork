<h1>Appending to a file in PHP</h1> 

<?php 

// Open the file for appending. If the file does not exist an 
// attempt will be made to create it. 

$out_file = @fopen("test.dat", "a") or die("Cannot open test.dat");   


fwrite($out_file, "This is line 1" . "\n"); 
fwrite($out_file, "This is line 2" . "\n"); 
fwrite($out_file, sin(1.0) . "\n"); 
fwrite($out_file, sin(2.0) . "\n"); 

// Close the output file 

fclose($out_file); 
?>
