<?php


function pig_latin($word)
{



$vowels = array('a', 'e', 'i', 'o', 'u');



if (in_array($word{0}, $vowels))

 {
return $word . 'way';
 } 

else 

{
return substr($word, 1) . $word{0} . 'ay';
}



}


echo pig_latin('hello'); // ellohay
echo '<br>'; // HTML line break
echo pig_latin('again'); // againway
?>
