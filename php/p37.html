<?php
// Display a table of trig functions for angles between 
// $start and $end in steps of $step degrees
function makeTable($start, $step, $end)
{
?>
   <table border="1" cellpadding="5">
   <tr>
   <th>x</th>
   <th>sin(x)</th>
   <th>cos(x)</th>
   <?php
   for ($x = $start; $x <= $end; $x += $step)
   {
      $a = deg2rad($x);
   ?>
   <tr>
      <td align="right"><?php printf("%.2f",$x) ?></td>
      <td><?php printf("%.5f",sin($a)) ?></td>
      <td><?php printf("%.5f",cos($a)) ?></td>
   </tr>
   <?php
   }
   ?>
   </table>
<?php
}
?>
