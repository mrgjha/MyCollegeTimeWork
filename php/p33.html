<h1>Reading a file in PHP</h1> 

<?php 
// Open the file for reading 

$in_file = @fopen("test.dat", "r") or die("Cannot open test.dat");   

// Read and display the lines 

while (! feof($in_file) ) 
{ 
   $line = fgets($in_file, 100); 
   echo $line, "<br>\n"; 
} 


// Close the output file 

fclose($in_file); 

?>
