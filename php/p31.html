<h1>Exploding and imploding strings in PHP</h1> 
<?php 

// explode converts a string to an array 
// based on a field separator 
// implode does the opposite 

$str_1 = "123:Fred Flintstone:450.55"; 
$array_1 = explode(":", $str_1); 
$str_2 = implode(":", $array_1); 

echo "<br>original string is $str_1"; 
echo "<br>exploded array is ", var_export($array_1); 
echo "<br>imploded string is $str_2"; 

?>
