<?php 

// Two forms using hidden fields to communicate between them. 
// Here we use GET method in forms so you can see query string 
// in the address bar. 

if ( isset($_REQUEST['page1']) ) 
{ 
   // First form was submitted so display page two 

   display_page_two(); 
} 
elseif ( isset($_REQUEST['page2']) ) 
{ 
   // Second form was submitted so display results page 

   display_result_page(); 
} 
else // No forms submitted yet so display page with first form 
{ 
   display_page_one(); 
} 

// -------------------------------------------------------- 

function display_page_one() 
{ 
   $self = $_SERVER['PHP_SELF'];    
?> 
   <html> 
   <head><title>Page 1</title></head> 
   <body> 
      <h1>Page 1</h1> 
   <form action="<?php echo $self ?>" method="GET"> 
      <table> 
      <tr> 
         <td>Name: </td> 
         <td><input type="text" name="name"></td> 
      </tr> 
      <tr> 
         <td>Age: </td> 
         <td><input type="text" name="age"></td> 
      </tr> 
      <tr> 
         <td><input type="submit" name="page1" value="Next Page"></td> 
      </tr> 
      </table> 
      </form> 
   </body> 
   </html> 
   <?php 
} 

function display_page_two() 
{ 
   // get the values just submitted from form one 

   $name = $_REQUEST['name']; 
   $age = $_REQUEST['age']; 
   $self = $_SERVER['PHP_SELF']; 

   // insert them in form 2 as hidden fields 

   ?> 
   <html> 
   <head><title>Page 2</title></head> 
   <body> 
   <h1>Page 2</h1> 
   <form action="<?php echo $self ?>" method="GET"> 
      <!-- START HIDDEN FIELDS --> 
      <input type="hidden" name="name" value="<?php echo $name ?>"> 
      <input type="hidden" name="age" value="<?php echo $age ?>"> 
      <!-- END HIDDEN FIELDS --> 
      <table> 
      <tr> 
         <td>Favourite book: </td> 
         <td><input type="text" name="book"></td> 
      </tr> 
      <tr> 
         <td><input type="submit" name="page2" value="Show Results"></td> 
      </tr> 
      </table> 
   </form> 
   </body> 
   </html> 
   <?php 
} 

function display_result_page() 
{ 
   // Form 2 was just submitted so get results for all fields 

   $name = $_REQUEST['name']; 
   $age = $_REQUEST['age']; 
   $book = $_REQUEST['book']; 
   ?> 
   <html> 
   <head><title>Results</title></head> 
   <body> 
   <h1>Results</h1> 
   <table border="1"> 
   <tr><td>Name</td><td><?php echo $name ?></td></tr> 
   <tr><td>Age</td><td><?php echo $age ?></td></tr> 
   <tr><td>Favourite book</td><td><?php echo $book ?></td></tr> 
   </table> 
   </body> 
   </html> 
   <?php 
}
