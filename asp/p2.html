<script language="javascript">

// Define constructor
function Circle(x, y, radius) {
  this.x = x
  this.y = y
  this.radius = radius
  this.display = display
  this.move = move
  this.scale = scale
}

// Define methods
function display() {
  alert("x = " + this.x + "; y = " + this.y + ""; radius = " + this.radius)
}

function move(x, y) {
  this.x = x
  this.y = y
}

function scale(a) {
  this.radius *= a
}

// Create a Circle object
circle = new Circle(0, 0, 1)

// Invoke methods
circle.scale(5)
circle.move(-2, -2)
circle.display()

</script>
