<script language="vbscript" runat="server">

Sub Application_OnStart
  Application("TotalUsers") = 0
  Application("CurrentUsers") = 0
End Sub

Sub Session_OnStart
  Application.Lock
  Application("TotalUsers") = Application("TotalUsers") + 1
  Application("CurrentUsers") = Application("CurrentUsers") + 1
  Application.UnLock
End Sub

Sub Session_OnEnd
  Application.Lock
  Application("CurrentUsers") = Application("CurrentUsers") - 1
  Application.UnLock
End Sub

</script>
