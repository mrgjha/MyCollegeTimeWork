<script language="vbscript">
Option Explicit
Dim data(3, 3), i, j, sum
For i = 0 To 3
  For j = 0 To 3
    data(i, j) = i * j
  Next
Next
For i = 0 To 3
  For j = 0 To 3
    sum = sum + data(i, j)
  Next
Next
MsgBox "The sum of all elements is " & sum
</script>

</script>

<script language="vbscript">
Option Explicit
Dim data(), i, sum

' Work with four element array
ReDim data(3)
For i = 0 To 3
  data(i) = i
Next
For i = 0 To 3
  sum = sum + data(i)
Next
MsgBox("Sum of digits 0-3 is " & sum)

' Expand it to a six element array
ReDim Preserve data(5)
For i = 4 To 5
  data(i) = i
Next
For i = 4 To 5
  sum = sum + data(i)
Next
MsgBox("Sum of digits 0-5 is " & sum)

</script>
