Private Sub Text1_KeyPress(KeyAscii As Integer)
  

' This fragment accepts only numerals
  'we will store the information in a
   'form-level variable I called fNumeral$
  

Static DecimalPointUsed As Integer

  Select Case KeyAscii

    Case Asc("0") To Asc("9")

      fNumeral$ = fNumeral$ + Chr$(KeyAscii)

    Case Asc(".")

      If DecimalPointUsed Then

        KeyAscii = 0

        Beep

       Else

        DecimalPointUsed = True

        fNumeral$ = fNumeral$ + Chr$(KeyAscii)
      End If
    Case Asc(",")
      
' Comma - do nothing to fNumeral$


    Case Else
      KeyAscii = 0
      Beep
  End Select
End Sub
