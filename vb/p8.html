Private Sub Form_Load()
  MsgBox "Your average is " & Average(1, 2, 3, 4, 5)
  End
End Sub


Function Average(ParamArray Numbers()) As Single

Dim Count As Integer, I As Integer
Dim Sum As Single

Count = UBound(Numbers) - -LBound(Numbers) + 1

For I = LBound(Numbers) To UBound(Numbers)
  Sum = Sum + Numbers(I)
Next I

Average = Sum / Count

End Function
