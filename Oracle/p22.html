Write a PL/SQL to show whether number is armstrong or not

 
--arm.sql



create or replace procedure armstrong(m in positive)
 is
 rem integer;
 x integer:=0;
 n integer;
 begin
 n:=m;
 while n>0
 loop
 rem:=mod(n,10);
 x:=x+rem*rem*rem;
 n:=floor(n/10);
 end loop;
 dbms_output.put_line('Sum of cubes of integer='||x);


 if x=m then
 dbms_output.put_line('Armstrong Number');
 else
 dbms_output.put_line('Not Armstrong');
 end if;
 end;

SQL> @arm.sql
SQL> exec armstrong(153)
