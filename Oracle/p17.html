Write a function to add two number using in parameter

 

create or replace function addnum(a in number,b in number)
return number
is
c number(3);
begin
c:=a+b;
return c;
end;


Calling Function 


SQL>declare
c number(3);
begin
c:=addnum(2,3);
dbms_output.put_line(c);
end;

SQL>exec dbms_output.put_line(addnum(2,3));
