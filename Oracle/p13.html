PL/SQL program to know the name of employee whose salary is minimum?

 create or replace procedure show_name
 is
 cursor name_cur is
 select ename from emp order by sal desc;
 rec name_cur%rowtype;
 begin
 open name_cur;
 for rows in 1..100
 loop
 fetch name_cur into rec;
 end loop;
 dbms_output.Put_Line(name_cur%rowcount);
 dbms_output.put_line(rec.ename);
 close name_cur;
 end;
