Create Tables with Constraints and Joining Them

SQL> create table Ordinals(
    Ordinal varchar2(10) primary key,
    Abbreviation varchar2(10
    )
    );

Table created.

SQL> create table Cardinals(
    Cardinal varchar2(10),
    Ordinal varchar2(10) references Ordinals(Ordinal),
    Abbreviation varchar2(10));

Table created.


SQL>insert into Ordinals values('&Ordinal','&Abbreviation')
  2  /

SQL> insert into Cardinals values('&Cardinal','&Ordinal','&Abbreviation')
  2  /

SQL> select * from Cardinals;

CARDINAL   ORDINAL    ABBREVIATI
---------- ---------- ----------

Two        Second     2
Three      Third      3
Four       Fourth     4
Five       Fifth      5

SQL> select * from Ordinals;

SQL> alter table Cardinals modify Abbreviation varchar2(15);

Table altered.

SQL> alter table Ordinals modify Abbreviation varchar2(15);
