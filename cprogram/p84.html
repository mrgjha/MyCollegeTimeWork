#include <stdio.h>
#include <graphics.h>
#include <conio.h>

int main(void)
{
  int driver, mode;
  struct palettetype p;

  driver = DETECT; /* autodetect */
  mode = 0;
  initgraph(&driver, &mode, "c:\\tc\\bgi");
  getpalette(&p);
  printf("number of colors in palette: %d", p.size);
  getch();
  restorecrtmode();
  return 0;
}
