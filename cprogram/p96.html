#include <graphics.h>
#include <conio.h>

int main(void)
{
  int driver, mode;
  /* define a fill pattern */
  char p[8] = {1, 2, 3, 4, 5, 6, 7};

  driver = DETECT; /* autodetect */
  mode = 0;
  initgraph(&driver, &mode, "c:\\tc\\bgi");

  setcolor(GREEN);
  rectangle(100, 200, 200, 300);

  setfillpattern(p, RED);
  floodfill(150, 250, GREEN);

  getch();
  restorecrtmode();
  return 0;
}
