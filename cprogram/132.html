 int k=0;

void ReadString(int x1,int y1,int endx,int color,int N,char *A)
{
settextstyle(SMALL_FONT,HORIZ_DIR,4);

char ch;

int i=0,ycur=y1+5,xcur=x1+2,h=textheight("Ip");
	do
	{
			while (!kbhit())
			 //If No Key Is Pressed Display A Cursor
			{
			setcolor(color);
			line(xcur+1,y1+3,xcur+1,y1+5+h);
			delay(150);
			setcolor(WHITE);
			line(xcur+1,y1+3,xcur+1,y1+5+h);
			delay(50);
			}

	ch=getch();
	if ( ch==13)//Enter Is Pressed
		     break;
		else if (ch=='\b' && i>0 )//BackSpace Pressed
		{
		setcolor(WHITE);
		outtextxy(x1+2,ycur,A);//Display Previous String In White
		A[--i]='\0';//Last Char Of String
		setcolor(color);
		outtextxy(x1+2,ycur,A);//Display New String
		xcur=x1+textwidth(A)+3;

		}
		else //Any Other Key is Pressed
		{
		setcolor(WHITE);
		outtextxy(x1+2,ycur,A);//Display Previous String in White
		A[i]=tolower(ch);
	       //	cout<<A[i];
		i++;//Push Ch into Array

		A[i]='\0';//Last char Of String

		setcolor(color);
		outtextxy(x1+2,ycur,A);//Display New Line
		xcur=x1+textwidth(A)+3;







		}



	} while ( i<N && xcur<endx );


	 int a=atoi(A);

	 k=k+a;





flushall();



}
