// Return a reference from [].
#include <iostream.h>


const int SIZE = 3;

class atype {
  int a[SIZE];
public:
atype();
int &operator[](int i) {return a[i];}
};
atype:: atype() {
    register int i;


  for(i=0; i<SIZE; i++) a[i] = i;
  }

int main()
{
  atype ob;

  cout << ob[2];  // displays 2
  cout << " ";

  ob[2] = 25;  // [] on left of =

  cout << ob[2];  // now displays 25

  return 0;
}
