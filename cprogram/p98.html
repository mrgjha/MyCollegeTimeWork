#include <graphics.h>
#include <conio.h>

int main(void)
{
  int driver, mode;
  int i;

  driver = DETECT; /* autodetect */
  mode = 0;
  initgraph(&driver, &mode, "c:\\tc\\bgi");

  for(i=0; i<4; i++) {
    setlinestyle(i, 0, 1);
    line(i*50, 100, i*50+50, 100);
  }

  getch();
  restorecrtmode();
  return 0;
}
