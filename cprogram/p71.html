#include <graphics.h>
#include <conio.h>

int main(void)
{
  int driver, mode;
  int shape[10] = { /* five points * 2 */
    10, 10,
    100, 80,
    200, 200,
    350, 90,
    0, 0
  };

  driver = DETECT; /* autodetect */
  mode = 0;
  initgraph(&driver, &mode, "c:\\bc5\\bgi");

  drawpoly(5, shape);
  getch();
  restorecrtmode();
  return 0;
}
