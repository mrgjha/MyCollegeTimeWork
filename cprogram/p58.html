#include <iostream.h>

class X {
protected:
  int a;
public:
  X() {
    a = 10;
    cout << "Initializing X\n";
  }
};

class Y {
protected:
  int b;
public:
  Y() {
    cout << "Initializing Y\n";
    b = 20;
  }
} ;

// Z inherits both X and Y
class Z : public X, public Y {
public:
  Z() { cout << "Initializing Z\n"; }
  int make_ab() { return a*b; }
} ;

int main()
{
  Z i;

  cout << i.make_ab();

  return 0;
}
