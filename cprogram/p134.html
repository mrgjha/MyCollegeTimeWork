#include<graphics.h>
#include<conio.h>
#include<dos.h>
#include<stdio.h>
void digital(void)
{

char buffer[80];
char c;

struct time t;
do{

delay(200);

clearviewport();
setcolor(WHITE);
gettime(&t);
sprintf(buffer, "%02d:%02d:%02d", t.ti_hour,t.ti_min,t.ti_sec);

settextstyle(GOTHIC_FONT, HORIZ_DIR, 5);
outtextxy(250,200,buffer);
}while(!kbhit());
getch();
closegraph();
}


void main()
{
int gdriver=DETECT,gmode;
initgraph(&gdriver,&gmode,"c:\\tc\\bgi");
digital();
}
