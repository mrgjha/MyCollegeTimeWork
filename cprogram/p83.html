#include <stdio.h>
#include <graphics.h>
#include <conio.h>

int main(void)
{
  int driver, mode;
  int high, low;

  driver = DETECT; /* autodetect */
  mode = 0;
  initgraph(&driver, &mode, "c:\\tc\\bgi");

  getmoderange(driver, &low, &high);
  printf("mode range: %d - %d", low, high);
  getch();
  restorecrtmode();
  return 0;
}
