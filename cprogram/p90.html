#include <graphics.h>
#include <conio.h>

int main(void)
{
  int driver, mode;
  int i, start, end;

  driver = DETECT; /* autodetect */
  mode = 0;
  initgraph(&driver, &mode, "c:\\tc\\bgi");

  /* demonstrate pieslice() */

  start = 0; end = 45;
  for(i=0; i<8; i++) {
    setfillstyle(SOLID_FILL, i);
    pieslice(300, 200, start, end, 100);
    start += 45;
    end += 45;
  }

  getch();
  restorecrtmode();
  return 0;
}
