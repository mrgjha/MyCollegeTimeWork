Sound

Importing sound will be done using 
the system speakers 
since they are almost allways the same 
across numerous computers 
and thus we don't have to program for 
different speaker cards.  
The system speakers are designed to 
emit a tone using it's 
corresponding hertz value. 
We will allocate the first octave 
of values in an array and then the 
rest will be assigned 
through the use of a loop.  The array will 
contain 13 notes and 8 octaves.  
For ease of use the notes 
will be defined to their corresponding values 
for example A is 0, 
A sharp (AS) is 1 and so forth.  
The values for octave 0 
are defined then the remaining 
8 octaves are assigned in the for loop.  
The two functions make_sound and make_rest 
do just what they say.  Before you exit 
the program you must have a nosound(); 
function or run through the make_rest 
function so the computer will stop making sound.

sound(65);
delay(1000);
nosound();

#include <dos.h>

#define A 0
#define AS 1
#define B 3
#define C 4
#define CS 5
#define D 6
#define DS 7
#define E 8
#define F 9
#define FS 10
#define G 11
#define GS 12

int make_sound(char N, int o, int d);
int make_rest(int l);
int notes[13][8];

//----------------------------------------------main-----------------------------------------
int main(void)
{
	notes[A][0] = 27;
	notes[AS][0] = 29;
	notes[B][0] = 31;
	notes[C][0] = 33;
	notes[CS][0] = 35;
	notes[D][0] = 37;
	notes[DS][0] = 39;
	notes[E][0] = 41;
	notes[F][0] = 44;
	notes[FS][0] = 46;
	notes[G][0] = 49;
	notes[GS][0] = 52;
//-------------------------assigns values to the remaining 8 octaves-------------------------
	for(int Z=1; Z<8; Z++)
	{
		notes[A][Z]=notes[A][0]*2;
		notes[AS][Z]=notes[AS][0]*2;
		notes[B][Z]=notes[B][0]*2;
		notes[C][Z]=notes[C][0]*2;
		notes[CS][Z]=notes[CS][0]*2;
		notes[D][Z]=notes[D][0]*2;
		notes[DS][Z]=notes[DS][0]*2;
		notes[E][Z]=notes[E][0]*2;
		notes[F][Z]=notes[F][0]*2*Z;
		notes[FS][Z]=notes[FS][0]*2*Z;
		notes[G][Z]=notes[G][0]*2*Z;
		notes[GS][Z]=notes[GS][0]*2*Z;
	}


	make_sound(A,8,1000);
	make_rest(1);
	
	return(0);
}

//-----------------------------------------make_sound----------------------------------------
int make_sound(char N, int o,int d)
{
	sound(notes[N][o]);
	delay(d);
	return 0;
}

//----------------------------------------make_rest------------------------------------------
int make_rest(int l)
{
	nosound();
	delay(l);
	return 0;
}
