#include <graphics.h>
#include <conio.h>

int main(void)
{
  int driver, mode;
  int i;

  driver = DETECT; /* autodetect */
  mode = 0;
  initgraph(&driver, &mode, "c:\\tc\\bgi");

  /* write two lines at CP */
  outtext("this is an example ");
  outtext("another line");

  /* use "cursor" positioning */
  for(i=100; i<200; i+=8) outtextxy(200, i, "hello");

  getch();
  restorecrtmode();
  return 0; 
}
