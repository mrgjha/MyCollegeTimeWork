#include <stdio.h>
#include <graphics.h>
#include <conio.h>

int main(void)
{
  int driver, mode;
  char f[8], num[10];
  int i;

  driver = DETECT; /* autodetect */
  mode = 0;
  initgraph(&driver, &mode, "c:\\tc\\bgi");
  getfillpattern((char far *) &f);

  /* display each byte in fill pattern */
  for(i=0; i<8; i++) {
    sprintf(num, "%d ", f[i]);
    outtext(num);
  }

  getch();
  restorecrtmode();
  return 0;
}
