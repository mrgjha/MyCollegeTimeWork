//Program to store BGI graphic file to disk 

#include<fstream.h> 
#include<graphics.h> 
#include<conio.h> 
void main() 
{ 
int a=DETECT,b; 
long int pixel,i,j; 
initgraph(&a,&b,"c:\\tc\\bgi"); 
/*IMAGE*/ 
setcolor(14); 
for(int ii=5;ii<=25;ii++) 
{circle(100,100,ii);ii++;} 
setcolor(13); 
for(ii=5;ii<=25;ii++) 
{circle(200,100,ii);ii++;} 
setcolor(12); 
for(ii=5;ii<=25;ii++) 
{circle(300,100,ii);ii++;} 
setcolor(11); 
for(ii=5;ii<=25;ii++) 
{circle(400,100,ii);ii++;} 
setcolor(10); 
for(ii=5;ii<=25;ii++) 
{circle(500,100,ii);ii++;} 

//save file to disk 
ofstream afile("pict");
for (i=0;i<=getmaxx();i++) 
    { 
    for(j=0;j<=getmaxy();j++) 
       { 
       pixel=getpixel(i,j); 
       afile<<pixel<<" "; 
       } 
    } 
afile.close(); 
//file is now saved 
cleardevice(); 
outtext("Press a key to continue..."); 
getch(); 
cleardevice(); 
//retrieve file 
ifstream file("pict"); 
for (i=0;i<=639;i++) 
    { 
    for(j=0;j<=479;j++) 
       { 
       file>>pixel; 
       putpixel(i,j,pixel); 
       } 
    } 
file.close(); 
outtext("File is now retrieved"); 
getch(); 
closegraph(); 
}
