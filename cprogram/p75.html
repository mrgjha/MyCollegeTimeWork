#include <graphics.h>
#include <conio.h>

int main(void)
{
  int driver, mode;

  driver = DETECT; /* autodetect */
  mode = 0;
  initgraph(&driver, &mode, "c:\\tc\\bgi");

  ellipse(100, 100, 0, 360, 80, 40);

  setfillstyle(XHATCH_FILL, MAGENTA);
  floodfill(100, 100, WHITE);

  getch();
  restorecrtmode();
  return 0;
}
