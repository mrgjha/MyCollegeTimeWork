union REGS i,o;
//Function for Getting mouse coordinates and button status
 getmousepos(int *button,int *x,int *y)
   {
     i.x.ax=3;
     int86(0x33,&i,&o);
     *button=o.x.bx;
     *x=o.x.cx;
     *y=o.x.dx;
      return 0;
   }

void Close_Button(int posx,int posy,int col,int shade1,int shade2)
{
     setfillstyle(1,col);
     bar(posx,posy,posx+12,posy+12);
     setcolor(shade1);
     line(posx-1,posy,posx-1,posy+13);
     line(posx-1,posy-1,posx+13,posy-1);
     setcolor(shade2);
     line(posx+13,posy-1,posx+13,posy+13);
     line(posx,posy+13,posx+13,posy+13);
     setcolor(0);
     settextstyle(2,0,5);
     outtextxy(posx+3,posy-2,"X");
}


void DrawWindow(int x1,int y1,int x2,int y2,int col,int barcolor,char *title)
{
	setfillstyle(1,col);      	//  Window Show on Screen
	bar(x1,y1,x2,y2);
	setfillstyle(1,7);              //  Title Bar
	bar(x1,y1+1,x2,y1+23);
	setcolor(7);                   // Side Lines
	bar(x1,y1+1,x1+3,y2);
	bar(x2-3,y1+1,x2,y2);
	setcolor(0);
	line(x1,y2+1,x2+1,y2+1);       // Under Shaded Lines  Black
	line(x1+4,y1+23,x1+4,y2-5);
	line(x1+4,y1+23,x2-4,y1+23);
	line(x1+3,y2-10,x2-3,y2-4);
	setcolor(15);
	line(x1,y1,x1,y2);             // Under Shaded Line  White
	line(x1,y1,x1,y2);
	line(x2-4,y1+23,x2-4,y2-5);
	line(x1+5,y2-4,x2-4,y2-4);
	setcolor(0);
	line(x2+1,y1,x2+1,y2);
	setfillstyle(1,7);       	//  Bottom Bar
	bar(x1+2,y2-18,x2,y2);
	setcolor(8);
	line(x1+3,y2-16,x1+3,y2-3);
	line(x1+3,y2-16,x2-4,y2-16);
	setcolor(15);
	line(x1+3,y2-3,x2-4,y2-3);
	line(x2-4,y2-16,x2-4,y2-3);
	setfillstyle(1,barcolor);       //  Top Bar
	bar(x1+3,y1+2,x2-3,y1+20);
	setcolor(15);            	//  Writting Text On Top Bar
	settextstyle(2,0,6);
	outtextxy(x1+3,y1+1,title);
	Close_Button(x2-19,y1+5,7,15,0);  // Call Function Buttons Close

}
