#include <graphics.h>
#include <conio.h>

int main(void)
{
  int driver, mode;
  driver = DETECT; /* autodetect */
  mode = 0;
  initgraph(&driver, &mode, "c:\\tc\\bgi");

  /* display a green 2-d bar */
  setfillstyle(SOLID_FILL, GREEN);
  bar(100, 100, 120, 200);

  /* now show a red 3-d bar */
  setfillstyle(SOLID_FILL, RED);
  bar3d(200, 100, 220, 200, 10, 1);

  getch();
  restorecrtmode();
  return 0;
}
