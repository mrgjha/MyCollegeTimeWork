Sub SubmitBtn_Click(Source As Object, E as EventArgs)
    MyMessage.InnerHTML = ""
    If chkDrawing.Checked = True Then
        MyMessage.InnerHTML = "You have entered the " _
            & "drawing.<BR>"
    End If
    If chkContact.Checked = True Then
        MyMessage.InnerHTML = MyMessage.InnerHTML _
            & "You are on our contact " _
            & "list.<BR>"
    End If
End Sub
