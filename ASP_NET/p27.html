Sub SubmitBtn_Click(Source As Object, E as EventArgs)

    If radBreakfast.Checked = True Then
        MyMessage.InnerHTML = "You selected " _
            & "breakfast.<BR>"

        MyMessage.InnerHTML = "You selected " _
            & "lunch.<BR>"

            & "dinner.<BR>"
    End If
End Sub
