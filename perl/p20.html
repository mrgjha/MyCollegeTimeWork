sub readCookies
{

my @cookie_values=split(/; /,$ENV{'HTTP_COOKIE'});


foreach (@cookie_values)
{

my ($cookie_name,$cookie_value)=split( /=/, $_);
$cookies{$cookie_name}=$cookie_value;
}

return %cookies;
}
