import javax.swing.*;
import java.awt.*;
public class Simple extends JApplet {
	public void init() {
		Container contentPane = getContentPane();
		Icon icon = new ImageIcon("balls10.gif", 
			"An animated GIF on a swing");
		JLabel label = new JLabel("Swing!", icon,
			SwingConstants.CENTER);
		contentPane.add(label, BorderLayout.CENTER);
	}
}



//JApplet has an instance of BorderLayout set for it’s 
//content pane.  BorderLayout.CENTER is default.
//Contrast: AWT’s Applet class uses FlowLayout.
