import java.applet.*;
import java.awt.*;
public class MyAppletSound  extends Applet {
        AudioClip ac; 
        public void init () {
                        		// load
          ac = getAudioClip(getDocumentBase(), "bburg1_1.mid");
        }
        public void start() {
          ac.loop();		// play
        }
        public void stop() {
          ac.stop(); 	// stop the sound upon leaving this web page
        }
} // end class MyAppletSound
