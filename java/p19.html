import javax.swing.JApplet;
import java.awt.*;

public class HelloApplet extends JApplet {
    	// paint can be used to draw an applet

    	public void paint(Graphics g) {
		g.drawString("Hello World", 25, 50);
    	} // paint
    
} // HelloApplet


//Note:This applet is non-GUI 
//and inherits from the Swing applet class, JApplet


<HTML><HEAD></HEAD>
	<BODY>
		<APPLET code="HelloApplet.class"
			   WIDTH="100" 
			   HEIGHT="100">
                Any text placed here shows in 
		non-Java enabled browsers	
		</APPLET>
	</BODY>
	</HTML>
//Note: Where the actual applet appears depends on the placement of the applet tag amongst the rest of the HTML – it is much like using an image tag.


/*
Create applet source file
HelloApplet.java
Create HTML source file
HelloApplet.html
Compile the java source, creating class file
javac HelloApplet.java
Load the html file
appletviewer HelloApplet.html

*/

//Note: the appletviewer is included with the JDK/SDK.  Acts basically as a mini-browser.
