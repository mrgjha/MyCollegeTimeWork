public class IncrementDecrementOperatorTest {
    public static void main(String [] args) {
	int count = 6, result;

	// increment operator

	count = 6;
	System.out.println("\ncount = " + count);
	count++; 
	System.out.println("count++; => count = " + count + "\n");
    
	count = 6;
	System.out.println("count = " + count);
	// result = count*++count*++count;
	// result = 6 * 7 * 8;
	result = count * (++count) * (++count);
	System.out.println("result = count*++count*++count; => " + 
			   "result = " + result +
			   ", count = " + count + "\n");

	count = 6;
	// result = count*count++*count++; 
	// result = 6 * 6 * 7;
	result = count * (count++) * (count++);
	System.out.println("count = " + count);
	System.out.println("result = count*count++*count++; => " + 
			   "result = " + result +
			   ", count = " + count + "\n");

	count = 6;
	// result = count++*count*count++; 
	// result = 7 * 6 * 7;
	result = (count++) * count * (count++);
	System.out.println("count = " + count);
	System.out.println("**result = count++*count*count++; => " + 
			   "result = " + result +
			   ", count = " + count + "\n");

	// decrement operator

	System.out.println("count = " + count);
	count--; 
	System.out.println("count--; => count = " + count + "\n");
    
	count = 6;
	// result = count*--count*--count;
	// result = 6 * 5 * 4;
	result = count * (--count) * (--count);
	System.out.println("count = " + count);
	System.out.println("result = count*--count*--count; => " + 
			   "result = " + result +
			   ", count = " + count + "\n");

	count = 6;
	// result = count*count--*count--;
	// result = 6 * 6 * 5;
	result = count * (count--) * (count--); 
	System.out.println("count = " + count);
	System.out.println("result = count*count--*count--; => " +
			   "result = " + result +
			   ", count = " + count);
    }
}
