import java.applet.*;
import java.awt.*;
public class MyimageApplet  extends Applet {
        Image im; 
        public void init () {
                        				// load
          im = getImage(getDocumentBase(),"gyan27.JPG");
          setBackground(Color.yellow);
        }
        public void paint(Graphics g ) {
            g.drawImage(im, 50, 50, this);	// display
        } 
} // end class MyApplet1
