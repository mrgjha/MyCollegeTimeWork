import javax.swing.event.*;
import javax.swing.*;

public class JTextAreaEventTest extends JCloseableFrame
    implements CaretListener {
    public JTextAreaEventTest() {
	super("JTextAreaEventTest (Swing)", 250, 100);

	JTextArea textArea = new JTextArea("My JTextArea");
	textArea.addCaretListener(this);
	
	getContentPane().add(textArea);
    }

    public void caretUpdate(CaretEvent event) {
	JTextArea textArea = (JTextArea)event.getSource();
	System.out.println("caretUpdate => " + textArea.getText());
    }

    public static void main(String [] args) {
	JTextAreaEventTest frame = new JTextAreaEventTest();
	frame.setVisible(true);
    }
}
