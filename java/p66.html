import java.awt.*;
import java.awt.event.*;
import javax.swing.*;

public class JCheckBoxEventTest extends JCloseableFrame
    implements ItemListener {
    public JCheckBoxEventTest() {
	super("JCheckBoxEventTest (Swing)", 220, 100);

	JPanel panel = new JPanel(new FlowLayout());

	JCheckBox checkBox = new JCheckBox("One");
	checkBox.addItemListener(this);
	panel.add(checkBox);
	
	checkBox = new JCheckBox("Two");
	checkBox.addItemListener(this);
	panel.add(checkBox);

	checkBox = new JCheckBox("Three");
	checkBox.addItemListener(this);
	panel.add(checkBox);
	
	setContentPane(panel);
    }

    public void itemStateChanged(ItemEvent event) {
	JCheckBox checkbox = (JCheckBox)event.getSource();
	System.out.println(checkbox.getText());
    }
    
    public static void main(String [] args) {
	JCheckBoxEventTest frame = new JCheckBoxEventTest();
	frame.setVisible(true);
    }
}
