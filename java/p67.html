import java.awt.*;
import java.awt.event.*;
import javax.swing.*;

public class JRadioButtonEventTest extends JCloseableFrame implements ItemListener {
    public JRadioButtonEventTest() {
	super("JRadioButtonEventTest (Swing)", 300, 100);

	JPanel panel = new JPanel(new FlowLayout());

	JRadioButton rb = new JRadioButton("One", false);
	rb.addItemListener(this);
	panel.add(rb);

	rb = new JRadioButton("Two", true);
	rb.addItemListener(this);
	panel.add(rb);

	rb = new JRadioButton("Three", false);
	rb.addItemListener(this);
	panel.add(rb);

	setContentPane(panel);
    }

    public void itemStateChanged(ItemEvent event) {
	JRadioButton radioButton = (JRadioButton)event.getSource();
	System.out.println(radioButton.getText());
    }

    public static void main(String [] args) {
	JRadioButtonEventTest frame = new JRadioButtonEventTest();
	frame.setVisible(true);
    }
}
