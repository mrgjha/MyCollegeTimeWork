import java.awt.event.*;
import javax.swing.*;

class WindowCB extends WindowAdapter {
    public void windowActivated(WindowEvent event) {
	System.out.println("windowActivated");
    }
}

public class JWindowAdapterTest extends JCloseableFrame {
    public JWindowAdapterTest(){
	super("JWindowAdapterTest (Swing)", 300, 100);
	addWindowListener(new WindowCB());
    }

    public static void main(String [] args) {
	JWindowAdapterTest frame = new JWindowAdapterTest();
	frame.setVisible(true);
    }
}
