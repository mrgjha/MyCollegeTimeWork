import java.awt.*;
import java.awt.event.*;

public class JMouseMotionListenerTest extends JCloseableFrame
    implements MouseMotionListener {
    public JMouseMotionListenerTest(){
	super("JMouseMotionListenerTest (Swing)", 300, 100);

	addMouseMotionListener(this);
    }

    // left key pressed & move
    public void mouseDragged(MouseEvent event) { 
	System.out.println("mouseDragged");
    }

    public void mouseMoved(MouseEvent event) {
	System.out.println("mouseMoved");
    }

    public static void main(String [] args) {
	JMouseMotionListenerTest frame = new JMouseMotionListenerTest();
	frame.setVisible(true);
    }
}
